create table USERACCOUNT (
USER_ID int NOT NULL AUTO_INCREMENT, 
USERNAME varchar(50),
PASSWORD varchar(50),
NAME varchar(50),
PRIMARY KEY (USER_ID)
);

create table TAGS_INFO (
TAG_ID int NOT NULL AUTO_INCREMENT,
TAG_NAME varchar(50),
PRIMARY KEY(TAG_ID)
);

create table TOPIC_DETAILS(
TOPIC_ID int NOT NULL AUTO_INCREMENT,
TOPIC_TITLE varchar(200),
TYPE int,
STATUS int,
CREATED_BY int REFERENCES USERACCOUNT(USER_ID),
PRIMARY KEY (TOPIC_ID)
);

create table TOPIC_TAG_MAPPING(
TOPIC_ID int REFERENCES TOPIC_DETAILS(TOPIC_ID),
TAG_ID int REFERENCES TAGS_INFO(TAG_ID)
);

create table USER_TAG_MAPPING(
USER_ID int REFERENCES USERACCOUNT(USER_ID),
TAG_ID int REFERENCES TAGS_INFO(TAG_ID)
);

create table TOPIC_MESSAGE_MAPPING(
TOPIC_ID int REFERENCES TOPIC_DETAILS(TOPIC_ID),
messages varchar(300)
);

create table TOPIC_USER_MAPPING(
TOPIC_ID int REFERENCES TOPIC_DETAILS(TOPIC_ID),
USER_ID int REFERENCES USERACCOUNT(USER_ID)
);
